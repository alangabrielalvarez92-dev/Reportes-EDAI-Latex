\section{Desarrollo}

\subsection{Algoritmo: Carrito de Compra con Estructura De Datos}

\begin{enumerate}
    \item Inicializar la lista de productos vacía.
    \item Mostrar el menú de opciones al usuario.
    \item Leer la opción seleccionada.
    \item Según la opción elegida, realizar una de las siguientes acciones:
    \begin{enumerate}
        \item Agregar un nuevo producto a la lista.
        \item Buscar un producto por su ID.
        \item Eliminar un producto de la lista.
        \item Mostrar el producto actual.
        \item Avanzar al siguiente producto.
        \item Retroceder al producto anterior.
        \item Mostrar todos los productos en orden hacia adelante.
        \item Mostrar todos los productos en orden hacia atrás.
        \item Salir del programa.
    \end{enumerate}
    \item Se libera la memoria utilizada.
    \item Se finaliza el programa mostrando el ticket final.
\end{enumerate}

\subsubsection{Descripción de Funciones}

\begin{itemize}

\item Función \textbf{crearNodo}: La función recibe tres argumentos: un entero \textbf{id}, una cadena \textbf{nombre} y un dato de tipo float \textbf{precio}. Su funcionamiento se divide en los siguientes pasos:
\begin{enumerate}
    \item Se reserva memoria dinámicamente para un nuevo nodo.
    \item Se verifica si la memoria fue asignada correctamente.
    \item Se asignan los valores de \textbf{id}, \textbf{nombre} y \textbf{precio} al nodo.
    \item Se inicializan los apuntadores \textbf{anterior} y \textbf{siguiente} en \textbf{NULL}.
    \item Se regresa el apuntador al nodo creado.
\end{enumerate}

\item Función \textbf{agregarProducto}: La función recibe el apuntador a la lista y los datos del producto. Su funcionamiento se divide en los siguientes pasos:
\begin{enumerate}
    \item Se crea un nuevo nodo.
    \item Se verifica si el nodo fue creado correctamente.
    \item Si la lista está vacía, el nodo se enlaza a sí mismo y se convierte en la cabeza.
    \item Si la lista no está vacía, se obtiene el último nodo.
    \item Se enlaza el nuevo nodo al final de la lista.
    \item Se actualizan los enlaces para mantener la estructura circular.
\end{enumerate}

\item Función \textbf{buscarProducto}: La función recibe la lista y un \textbf{id}. Su funcionamiento se divide en los siguientes pasos:
\begin{enumerate}
    \item Se verifica si la lista está vacía.
    \item Se recorre la lista con un apuntador auxiliar.
    \item Se compara cada \textbf{id} con el valor buscado.
    \item Si se encuentra coincidencia, se regresa el nodo.
    \item Si no se encuentra, se regresa \textbf{NULL}.
\end{enumerate}

\item Función \textbf{eliminarProducto}: La función recibe la lista y un \textbf{id}. Su funcionamiento se divide en los siguientes pasos:
\begin{enumerate}
    \item Se verifica si la lista está vacía.
    \item Se busca el nodo a eliminar.
    \item Si no existe, la función termina.
    \item Si solo hay un nodo, se elimina y la lista queda vacía.
    \item Si hay más nodos, se actualizan los enlaces de los nodos vecinos.
    \item Si el nodo eliminado es la cabeza, se actualiza al siguiente nodo.
    \item Se libera la memoria del nodo eliminado.
\end{enumerate}

\item Función \textbf{mostrarProductoActual}: La función recibe un nodo actual. Su funcionamiento se divide en los siguientes pasos:
\begin{enumerate}
    \item Se verifica si el nodo es \textbf{NULL}.
    \item Si no lo es, se muestran los datos del producto.
\end{enumerate}

\item Función \textbf{siguienteProducto}: La función recibe un nodo actual. Su funcionamiento se divide en los siguientes pasos:
\begin{enumerate}
    \item Se verifica que el nodo no sea \textbf{NULL}.
    \item Se avanza al siguiente nodo.
    \item Se muestra el producto actual.
\end{enumerate}

\item Función \textbf{anteriorProducto}: La función recibe un nodo actual. Su funcionamiento se divide en los siguientes pasos:
\begin{enumerate}
    \item Se verifica que el nodo no sea \textbf{NULL}.
    \item Se retrocede al nodo anterior.
    \item Se muestra el producto actual.
\end{enumerate}

\item Función \textbf{mostrarProductos}: La función recibe la lista. Su funcionamiento se divide en los siguientes pasos:
\begin{enumerate}
    \item Se verifica si la lista está vacía.
    \item Se inicia el recorrido desde el primer nodo.
    \item Se muestra el producto actual.
    \item Se presenta un menú de navegación.
    \item Se avanza o retrocede según la opción elegida.
    \item Si el usuario decide salir, la función termina.
\end{enumerate}

\item Función \textbf{mostrarTicket}: La función recibe la lista. Su funcionamiento se divide en los siguientes pasos:
\begin{enumerate}
    \item Se verifica si la lista está vacía.
    \item Se recorre la lista desde el primer nodo.
    \item Se imprimen los datos de cada producto.
\end{enumerate}

\item Función \textbf{mostrarTicketAdelante}: La función recibe la lista. Su funcionamiento se divide en los siguientes pasos:
\begin{enumerate}
    \item Se verifica si la lista está vacía.
    \item Se recorre la lista en orden hacia adelante.
    \item Se imprimen los datos de cada producto.
\end{enumerate}

\item Función \textbf{mostrarTicketAtras}: La función recibe la lista. Su funcionamiento se divide en los siguientes pasos:
\begin{enumerate}
    \item Se verifica si la lista está vacía.
    \item Se inicia desde el último nodo.
    \item Se recorre la lista hacia atrás.
    \item Se imprimen los datos de cada producto.
\end{enumerate}

\item Función \textbf{liberarLista}: La función recibe la lista. Su funcionamiento se divide en los siguientes pasos:
\begin{enumerate}
    \item Se verifica si la lista está vacía.
    \item Se recorren los nodos liberando la memoria de cada uno.
    \item Se libera el nodo inicial.
    \item Se asigna \textbf{NULL} a la lista.
\end{enumerate}

\item Función \textbf{mostrarMenu}: Le muestra al usuario las opciones del programa.

\end{itemize}
